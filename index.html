Dim DbObat As ADODB.Connection
Dim RsObat As ADODB.Recordset


Private Sub Form_Load()
Set DbObat = New ADODB.Connection
DbObat.Provider = "MICROSOFT.JET.OLEDB.4.0"
DbObat.CursorLocation = adUseClient
DbObat.Open "D:\DataObat.mdb"
Set RsObat = New ADODB.Recordset
RsObat.Open "Obat", DbObat, 2, 3
Set DataGrid1.DataSource = RsObat
End Sub


Sub kosong()
t1 = ""
t2 = ""
t3 = ""
t4 = ""
t5 = ""
t6 = ""
End Sub


Private Sub c1_Click()
RsObat.AddNew
